// REGLAS DE FIRESTORE - Agregar estas reglas a tu Firebase Console

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // Reglas existentes para user_profiles
    match /user_profiles/{userId} {
      allow read, write: if true;
    }
    
    // Reglas existentes para attendance_records
    match /attendance_records/{recordId} {
      allow read, write: if true;
    }
    
    // Reglas existentes para group_enrollments
    match /group_enrollments/{enrollmentId} {
      allow read, write: if true;
    }
    
    // Reglas existentes para events
    match /events/{eventId} {
      allow read, write: if true;
    }
    
    // Reglas existentes para event_attendance_records
    match /event_attendance_records/{recordId} {
      allow read, write: if true;
    }
    
    // NUEVAS REGLAS - Agregar estas colecciones
    
    // Usuarios administradores
    match /admin_users/{adminId} {
      allow read, write: if true;
    }
    
    // Encargados de grupos (directores/monitores)
    match /group_managers/{managerId} {
      allow read, write: if true;
    }
    
    // Asignaciones de categorías (semillero, proceso, representativo)
    match /group_category_assignments/{assignmentId} {
      allow read, write: if true;
    }
  }
}
